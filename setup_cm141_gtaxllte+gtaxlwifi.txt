vim ~/lunch_gtaxllte.sh
--------------------------------------------
#!/bin/bash

cd ~/android/system
repo sync --force-sync
. build/envsetup.sh
breakfast gtaxllte
brunch gtaxllte

echo "gtaxllte" | mail -s "gtaxllte build done!" $BUILD_DONE_EMAIL
cat ~/android/system/build/core/version_defaults.mk | grep --color "PLATFORM_SECURITY_PATCH :="
caja ~/android/system/out/target/product/gtaxllte
--------------------------------------------
chmod +x ~/lunch_gtaxllte.sh
